│                                                                                                                                                                                                                                                                    │
│ Context                                                                                                                                                                                                                                                            │
│                                                                                                                                                                                                                                                                    │
│ Build a full-stack tool that accepts an Interactive Brokers Activity Statement (CSV) or Flex Query (XML) export, extracts dividends/withholding tax and interest income, maps the data to a strict eCH-0196 v2.2 XML document, embeds the full XML in a PDF417     │
│ barcode, and generates a downloadable PDF. Institution-specific fields (clearing number, customer number, canton) are provided via a configuration form in the UI.                                                                                                 │
│                                                                                                                                                                                                                                                                    │

│ 3. IB File Parser (backend/src/main/kotlin/com/steuerauszug/backend/parser/)                                                                                                                                                                                       │
│                                                                                                                                                                                                                                                                    │
│ IbParser.kt — interface:                                                                                                                                                                                                                                           │
│ interface IbParser {                                                                                                                                                                                                                                               │
│     fun parse(content: String): IbActivityData                                                                                                                                                                                                                     │
│ }                                                                                                                                                                                                                                                                  │
│                                                                                                                                                                                                                                                                    │
│ IbCsvParser.kt — handles Activity Statement CSV:                                                                                                                                                                                                                   │
│ - Reads the multi-section CSV line by line                                                                                                                                                                                                                         │
│ - Groups rows by section header (first column)                                                                                                                                                                                                                     │
│ - Extracts "Dividends" section → IbDividend list                                                                                                                                                                                                                   │
│ - Extracts "Withholding Tax" section → IbWithholdingTax list (amounts are negative in IB CSV, negate them)                                                                                                                                                         │
│ - Extracts "Interest" section → IbInterest list                                                                                                                                                                                                                    │
│ - Matches dividends with their corresponding withholding tax by symbol                                                                                                                                                                                             │
│                                                                                                                                                                                                                                                                    │
│ IbFlexQueryParser.kt — handles Flex Query XML:                                                                                                                                                                                                                     │
│ - Parses <FlexQueryResponse> XML                                                                                                                                                                                                                                   │
│ - Extracts <CashTransaction> elements with type Dividends, Withholding Tax, BrokerInterest                                                                                                                                                                         │
│                                                                                                                                                                                                                                                                    │
│ 4. Mapper (backend/src/main/kotlin/com/steuerauszug/backend/mapper/IbToEchMapper.kt)                                                                                                                                                                               │
│                                                                                                                                                                                                                                                                    │
│ - Converts IbActivityData + GenerationRequest → EchTaxStatement                                                                                                                                                                                                    │
│ - Groups dividends by security, nets withholding tax against each                                                                                                                                                                                                  │
│ - Generates a unique documentId: CH-{clearingNumber}-{customerNumber}-{taxYear}-001                                                                                                                                                                                │
│ - Sets periodFrom = {taxYear}-01-01, periodTo = {taxYear}-12-31                                                                                                                                                                                                    │
│ - Aggregates totals                                                                                                                                                                                                                                                │
│                                                                                                                                                                                                                                                                    │
│ 5. XML Generator (backend/src/main/kotlin/com/steuerauszug/backend/generator/EchXmlGenerator.kt)                                                                                                                                                                   │
│                                                                                                                                                                                                                                                                    │
│ - Builds eCH-0196 v2.2 compliant XML string using javax.xml.stream.XMLStreamWriter                                                                                                                                                                                 │
│ - Namespace: http://www.ech.ch/xmlns/eCH-0196/2                                                                                                                                                                                                                    │
│ - Root element taxStatement with attributes: taxPeriod, periodFrom, periodTo, canton, totalTaxValue                                                                                                                                                                │
│ - Child elements: administrativeInformation, items (one item per TaxItem), totalValues                                                                                                                                                                             │
│ - Result is a UTF-8 XML string that will be embedded in the PDF417 barcode AND attached as metadata                                                                                                                                                                │
│                                                                                                                                                                                                                                                                    │
│ 6. PDF Generator (backend/src/main/kotlin/com/steuerauszug/backend/generator/PdfGenerator.kt)                                                                                                                                                                      │
│                                                                                                                                                                                                                                                                    │
│ Barcode generation (ZXing):                                                                                                                                                                                                                                        │
│ val hints = mapOf(EncodeHintType.CHARACTER_SET to "UTF-8", EncodeHintType.ERROR_CORRECTION to 2)                                                                                                                                                                   │
│ val writer = PDF417Writer()                                                                                                                                                                                                                                        │
│ val bitMatrix = writer.encode(xmlContent, BarcodeFormat.PDF_417, 800, 400, hints)                                                                                                                                                                                  │
│ val barcodeImage = MatrixToImageWriter.toBufferedImage(bitMatrix)                                                                                                                                                                                                  │
│                                                                                                                                                                                                                                                                    │
│ PDF layout (iText 7):                                                                                                                                                                                                                                              │
│ - A4 page landscape                                                                                                                                                                                                                                                │
│ - Header: "E-Steuerausweis {taxYear}" + document ID + canton                                                                                                                                                                                                       │
│ - Institution / customer info block (two-column table)                                                                                                                                                                                                             │
│ - Income items table: Description | Currency | Gross | Withholding Tax | Net                                                                                                                                                                                       │
│ - Totals row                                                                                                                                                                                                                                                       │
│ - PDF417 barcode image embedded at the bottom                                                                                                                                                                                                                      │
│ - Footer note: "Generiert gemäss eCH-0196 v2.2.0"                                                                                                                                                                                                                  │
│                                                                                                                                                                                                                                                                    │
│ 7. REST Controller (backend/src/main/kotlin/com/steuerauszug/backend/controller/TaxStatementController.kt)                                                                                                                                                         │
│                                                                                                                                                                                                                                                                    │
│ @RestController                                                                                                                                                                                                                                                    │
│ @RequestMapping("/api/steuerausweis")                                                                                                                                                                                                                              │
│ @CrossOrigin(origins = ["http://localhost:4200"])                                                                                                                                                                                                                  │
│ class TaxStatementController(                                                                                                                                                                                                                                      │
│     private val ibCsvParser: IbCsvParser,                                                                                                                                                                                                                          │
│     private val ibFlexQueryParser: IbFlexQueryParser,                                                                                                                                                                                                              │
│     private val mapper: IbToEchMapper,                                                                                                                                                                                                                             │

│     private val pdfGenerator: PdfGenerator                                                                                                                                                                                                                         │
│ ) {                                                                                                                                                                                                                                                                │
│     @PostMapping("/generate", consumes = [MediaType.MULTIPART_FORM_DATA_VALUE])                                                                                                                                                                                    │
│     fun generate(                                                                                                                                                                                                                                                  │
│         @RequestPart("file") file: MultipartFile,                                                                                                                                                                                                                  │
│         @RequestPart("request") request: GenerationRequest                                                                                                                                                                                                         │
│     ): ResponseEntity<ByteArray> {                                                                                                                                                                                                                                 │
│         val content = file.inputStream.bufferedReader().readText()                                                                                                                                                                                                 │

│                      else ibCsvParser.parse(content)                                                                                                                                                                                                               │
│         val statement = mapper.map(ibData, request)                                                                                                                                                                                                                │

│         val pdf = pdfGenerator.generate(statement, xml)                                                                                                                                                                                                            │
│         return ResponseEntity.ok()                                                                                                                                                                                                                                 │
│             .header(HttpHeaders.CONTENT_DISPOSITION, "attachment; filename=steuerausweis-${request.taxYear}.pdf")                                                                                                                                                  │
│             .contentType(MediaType.APPLICATION_PDF)                                                                                                                                                                                                                │
│             .body(pdf)                                                                                                                                                                                                                                             │
│     }                                                                                                                                                                                                                                                              │
│ }                                                                                                                                                                                                                                                                  │
